select OBID, OBFUID FUID
, OBTXNENTITY ENTITY, OBTXNTYPE , OBDATE OBLIDATE, OBTYPE TYPE, OBSTATUS STATUS, OBAMOUNT AMT, OBORIGINALAMOUNT ORIGINALAMT
, OBSPARTNUMBER PART, OBSSTATUS SPLITSTATUS, OBSSETTLEDAMOUNT SETTLMENTAMT, OBSRESPREMARKS REMARKS, OBSPAYMENTSETTLOR SETTLOR
 from obligations, obligationsplits
where obsobid = obid
AND OBRECORDVERSION > 0 AND OBSRECORDVERSION > 0
AND OBSPAYMENTSETTLOR = 'DIRECT'
AND OBSTATUS in( 'RDY' , 'CRT', 'SNT')
