SELECT FUPURCHASER
, FUSUPPLIER
, FUSTATUS
, FUFACTOREDAMOUNT
, FUFINANCIER
, FUACCEPTDATETIME
, CoPurc.CDCONSTITUTION 
,(CASE WHEN CoPurc.CDCONSTITUTION='PROP' THEN 'PROPRITORYSHIP' 
WHEN CoPurc.CDCONSTITUTION= 'PRIV' THEN 'PRIVATE'
WHEN CoPurc.CDCONSTITUTION= 'PUB' THEN 'PUBLIC'
WHEN CoPurc.CDCONSTITUTION= 'PART' THEN 'PARTNERSHIP'
WHEN CoPurc.CDCONSTITUTION= 'TRST' THEN 'TRUST'
WHEN CoPurc.CDCONSTITUTION= 'HUF' THEN 'HUF'
ELSE '' END)AS PurConstitution
, CoSupp.CDMSMESTATUS
, CoFin.CDFINANCIERCATEGORY

 FROM FACTORINGUNITS
 , COMPANYDETAILS CoPurc
 , COMPANYDETAILS CoSupp
 , COMPANYDETAILS CoFin
WHERE FURECORDVERSION > 0
AND CoPurc.CDCODE = FuPurchaser
AND CoSupp.CDCODE = FuSupplier
AND CoFin.CdCode = Fufinancier
