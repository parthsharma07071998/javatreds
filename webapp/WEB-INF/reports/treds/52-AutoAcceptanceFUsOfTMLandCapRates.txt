SELECT FUID, fustatus STATUS, FUSUPPLIER SUPPLIER, FUOWNERENTITY OWNER, FUACCEPTINGENTITY ACCEPTINGENTITY
, fubdid BIDID
, BDFINANCIERENTITY FINANCIER
, bdrate BIDRATE
, PSLENTITYCODE	ENTITYCODE, PSLCOUNTERENTITYCODE COUNTER
, PSLFROMHAIRCUT FROMHAIRCUT
, PSLTOHAIRCUT TOHAIRCUT
, PSLFROMUSANCE FROMUSANCE
, PSLTOUSANCE TOUSANCE
, PSLCAPRATE CAPRATE
 FROM FACTORINGUNITS 
LEFT OUTER JOIN PURCHASERSUPPLIERCAPRATE ON (PSLENTITYCODE= FUSUPPLIER AND PSLCOUNTERENTITYCODE = 'TA0001023' )
left outer join bids on (fubdid = bdid)
WHERE  FUPURCHASER = 'TA0001023' 
AND FUSTATUS='ACT'
and fuenabLEEXTENSION ='Y'
ORDER BY PSLENTITYCODE DESC
