SELECT * FROM COMPANYCONTACTS, COMPANYDETAILS
WHERE CDRECORDVERSION > 0 AND CCRECORDVERSION > 0
AND CCCDID = CDID 
AND ccnoaemail is null and ccadmin = 'Y'
AND CDCODE IN (
SELECT DISTINCT ANPURCHASER ENTITY FROM ASSIGNMENTNOTICES
UNION
SELECT DISTINCT ANSUPPLIER ENTITY FROM ASSIGNMENTNOTICES
UNION
SELECT DISTINCT ANFINANCIER ENTITY FROM ASSIGNMENTNOTICES
)
