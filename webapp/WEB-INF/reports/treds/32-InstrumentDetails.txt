select instrumentworkflow.* , INpURCHASER, INSUPPLIER
from instrumentworkflow 
LEFT OUTER JOIN Instruments ON ( IWFInid = Inid  )
WHERE NVL(TO_NUMBER(TO_CHAR(IWFSTATUSUPDATETIME, 'YYYYMMDD')),0) >= 20181101
AND Inpurchaser in ( 'BA0000047','BE0000187','CG0000519','EA0000032','FU0000508','GI0000498','GR0000177','HI0000289','HP0000100','IO0000268','JA0000267','NA0000063','VH0000285','VI0000067')
ORDER BY IWFSTATUSUPDATETIME DESC
