 SELECT (CASE WHEN FASLEVEL ='YNNN' THEN 'Financier Self'
WHEN FASLEVEL ='YYNN' THEN 'Financier Buyer'
WHEN FASLEVEL ='YYYN' THEN 'Financier Buyer Seller'
WHEN FASLEVEL ='YNNY' THEN 'Financier User'
WHEN FASLEVEL ='NYNN' THEN 'System Buyer'
ELSE '' END) AS "FASLEVEL",FASLEVEL, FASFINANCIER, FASPURCHASER, FASPURCHASERREF, FASSUPPLIER, FASLIMIT,  FASBASERATE, FASMINBIDRATE, FASMAXBIDRATE, FASMINSPREAD, FASMAXSPREAD, FASBIDLIMIT, FASPURCHASERCOSTLEG, FASACTIVE
FROM FINANCIERAUCTIONSETTINGS
