select OBFUID FuId, OEPURCHASER Purchaser, OEFINANCIER Financier
, CASE WHEN OESTATUS='P' THEN 'Pending' 
  WHEN OESTATUS='A' THEN 'Approved' 
  WHEN OESTATUS='R' THEN 'Rejected' 
  WHEN OESTATUS='E' THEN 'Expired'
  ELSE '' END Status
, OEOLDDATE OLDDATE, OENEWDATE NEWDATE
, OEOLDAMOUNT OLDAMOUNT
, OEINTEREST INTEREST, OEPENALTY PENALTY, OEPENALTYRATE PENALTYRATE
, OENEWAMOUNT NEWAMOUNT
, OEREMARKS REMARKS
  from OBLIGATIONEXTENSIONS, Obligations
WHERE ObRecordVersion > 0
AND OEOBID = OBID
AND OENEWDATE >= SYSDATE
ORDER BY OEPURCHASER, OEFINANCIER, OESTATUS
